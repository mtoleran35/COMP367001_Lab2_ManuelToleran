pipeline {
    agent any
	
	//Stages
    stages {
        // Checkout Stage
        stage('Checkout') {
            steps {
				checkout scm 
            }
        }

        // Build Stage
        stage('Build') {
            steps {
				script {
                    sh 'make'
                }
                echo 'Building application...'
            }
        }

        // Test Stage
        stage('Test') {
            steps {
                echo 'Testing application...'
            }
        }

        // Deploy Stage
        stage('Deploy') {
            steps {
                echo 'Deploying application...'
            }
        }
    }
	
	post {
        success {
            // Archive files after successful build
            archiveArtifacts artifacts: '**/target/*.jar', allowEmptyArchive: true
        }
    }

}
