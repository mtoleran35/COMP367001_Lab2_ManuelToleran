pipeline {
    agent any
	
	environment{
		PATH = "C:/Program Files/Docker/Docker:$PATH"
		DOCKERHUB_PWD = credentials('CredentialID_DockerHubPWD')
	}
	
	//Stages
    stages {
        // Checkout Stage
        stage('Checkout') {
            steps {
				checkout scm 
            }
        }

        // Build Maven Stage
        stage('Build Maven') {
            steps {
				script {
                    sh 'mvn -f ManuelToleran_COMP367001_Lab2_Q2/pom.xml install'
                }
                echo 'Building application...'
            }
        }
		
		// Login Docker
        stage('Docker Login') {
            steps {
				script {
                    sh 'docker login -u mtoleran -p ${DOCKERHUB_PWD}'
                }
                echo 'Logging in to Docker...'
            }
        }
		
		// Build Docker Image Stage
        stage('Build Docker Image') {
            steps {
				script {
                    sh 'docker build -t mtoleran/manueltoleran-lab3 .'
                }
                echo 'Building docker image...'
            }
        }

    }
}
